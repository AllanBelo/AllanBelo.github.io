Let’s create an interface for the class PeopleRepository (soon we will create it) in the Repository folder, create the interface and name it IPeopleRepository (it’s a good practice naming the inferface this way): 
using ApiDotNetCore.Model;
namespace ApiDotNetCore.Repository
{
    public interface IPeopleRepository
    {
        //Metodos impostos para classe que implementam a interface
         public People GetPeople(int PersonID);
    }
}
Also in the Repository folder, create the class PeopleRepository (that’s our most complex class): 
using ApiDotNetCore.Model;
using MySqlConnector;

namespace ApiDotNetCore.Repository
{
    public class PeopleRepository : IPeopleRepository
    {
        //Variable string BD connection
        private readonly string _connectionString;
        //Connection string in the class constructor
        public PeopleRepository(string connectionString)
        {
            _connectionString = connectionString;
        }
        //Implementation o the method imposed by the interface
        People IPeopleRepository.GetPeople(int PersonID)
        {
            using(MySqlConnection connection = new MySqlConnection(_connectionString))
            {          
                People people = null;   
                MySqlCommand query = new MySqlCommand();
                //Opening Connection with the bank
                connection.Open(); 
                //Assigning connection opened to query.Connection
                query.Connection = connection; 
                //Consulta
                query.CommandText = @"SELECT PersonID, FirstName, LastName, Age FROM People WHERE PersonID =@PersonID"; //Atribuindo consulta a query.CommandText
                //Good practice to avoid SQL Injection
                query.Parameters.AddWithValue("@PersonID", PersonID); 
                //Executing to query
                MySqlDataReader reader = query.ExecuteReader(); 
                //Assembling Object with the result of the consult
                while(reader.Read())
                {
                    people = new People(
                    (int)reader["PersonID"], 
                    (string)reader["FirstName"],
                    (string)reader["LastName"],
                    (int)reader["Age"]
                    );
                }
                return people;               
            }
        }
    }
}
We need to create the connection string with the data bank, open the appsettings.json file and create the string:

Now let’s pass the string for the class PeopleRepository, open the class Startup and look for the method ConfigureServices and do this: 
 public void ConfigureServices(IServiceCollection services)
 {           
    services.AddControllers();
    //Passing the string connection for the class PeopleRepository
    services.AddScoped<IPeopleRepository>(factory => {
    return new PeopleRepository(
        Configuration.GetConnectionString("MySqlDbConnection")
        );
     });
 }
Inside the Controllers folder create the class PeopleController: 
using ApiDotNetCore.Repository;
using Microsoft.AspNetCore.Mvc;

namespace ApiDotNetCore.Controllers
{
    [ApiController]
    [Route("/api/[controller]")]
    public class PeopleController:ControllerBase
    {
        private readonly IPeopleRepository _peopleRepository;
        public PeopleController(IPeopleRepository peopleRepository)
        {
            _peopleRepository = peopleRepository;
        }
        [HttpGet("{PersonID}")]
        public IActionResult GetAction(int PersonID)
        {
            var people = _peopleRepository.GetPeople(PersonID);
            if(people == null)
            {
                return NotFound();
            }
            return Ok(people);
        }
    }
}
